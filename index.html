  <script>

    const LS_KEY = "expenses_v1";


    function todayISO() {

      const d = new Date();

      const m = String(d.getMonth()+1).padStart(2,"0");

      const day = String(d.getDate()).padStart(2,"0");

      return `${d.getFullYear()}-${m}-${day}`;

    }


    function readExpenses() {

      try { return JSON.parse(localStorage.getItem(LS_KEY)) || []; }

      catch { return []; }

    }

    function saveExpenses(list) { localStorage.setItem(LS_KEY, JSON.stringify(list)); }


    function formatMoney(n) {

      const num = Number(n) || 0;

      return num.toLocaleString(undefined, {minimumFractionDigits: 2, maximumFractionDigits: 2});

    }


    function render() {

      const tbody = document.getElementById("tbody");

      const totalCell = document.getElementById("totalCell");

      const list = readExpenses();


      tbody.innerHTML = "";

      let total = 0;

      list.forEach((e, idx) => {

        total += Number(e.amount) || 0;

        const tr = document.createElement("tr");

        tr.innerHTML = `

          <td>${e.date}</td>

          <td>${e.category}</td>

          <td>${e.desc || ""}</td>

          <td>${formatMoney(e.amount)}</td>

          <td>

            <button data-del="${idx}" title="Delete">Delete</button>

          </td>

        `;

        tbody.appendChild(tr);

      });

      totalCell.textContent = formatMoney(total);


      // Wire delete buttons

      tbody.querySelectorAll("button[data-del]").forEach(btn => {

        btn.addEventListener("click", () => {

          const idx = Number(btn.getAttribute("data-del"));

          const list = readExpenses();

          list.splice(idx, 1);

          saveExpenses(list);

          render();

        });

      });

    }


    function addExpense() {

      const date = document.getElementById("date").value || todayISO();

      const category = document.getElementById("category").value.trim() || "Other";

      const desc = document.getElementById("desc").value.trim();

      const amount = document.getElementById("amount").value;


      if (!amount || Number(amount) <= 0) {

        alert("Please enter a valid amount.");

        return;

      }


      const list = readExpenses();

      list.push({ date, category, desc, amount: Number(amount) });

      saveExpenses(list);

      document.getElementById("desc").value = "";

      document.getElementById("amount").value = "";

      render();

    }


    function clearAll() {

      if (confirm("This will remove all expenses saved in your browser. Continue?")) {

        localStorage.removeItem(LS_KEY);

        render();

      }

    }


    function exportCSV() {

      const rows = readExpenses();

      if (!rows.length) { alert("No data to export."); return; }

      const header = ["Date","Category","Description","Amount"];

      const csv = [header.join(",")].concat(

        rows.map(r => [r.date, r.category, (r.desc || "").replace(/"/g,'""'), r.amount].map(x => `"${x}"`).join(","))

      ).join("\n");


      const blob = new Blob([csv], { type: "text/csv;charset=utf-8;" });

      const url = URL.createObjectURL(blob);

      const a = document.createElement("a");

      a.href = url;

      a.download = `expenses_${new Date().toISOString().slice(0,10)}.csv`;

      a.click();

      URL.revokeObjectURL(url);

    }


    // Init

    document.getElementById("date").value = todayISO();

    document.getElementById("addBtn").addEventListener("click", addExpense);

    document.getElementById("clearAllBtn").addEventListener("click", clearAll);

    document.getElementById("exportBtn").addEventListener("click", exportCSV);

    render();

  </script>
